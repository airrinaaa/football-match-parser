6///Parser for football match lines based on the grammar defined in grammar.pest
7#[derive(Parser)]
8#[grammar = "src/grammar.pest"]
9pub struct MatchGrammar;

11///Represents one football match with parsed data from the input line
12#[derive(Debug, PartialEq)]
13pub struct Match {
14    ///The date of the match in YYYY-MM-DD format
15    pub date: String,
16    /// The match start time in HH:MM format
17    pub time: String,
18    ///The home team name
19    pub home_team: String,
20    ///The away team name
21    pub away_team: String,
22    ///The home team's score(if the match has already been played)
23    pub home_score: Option<u8>,
24    ///The away team's score(if the match has already been played)
25    pub away_score: Option<u8>,
26    ///The stadium where the match is taking place
27    pub stadium: Option<String>,
28    ///The current status of the match(played, ongoing, scheduled)
29    pub status: Option<String>,
30}
31
32///Custom error type
33#[derive(Error, Debug)]
34pub enum ParseError {
35    ///Syntax - when a line doesn't match grammar rules  
36    #[error("Syntax error while parsing line: {0}")]
37    Syntax(String),
38    ///FieldFormat - when a field has an invalid format(for ex. date doesn't exist)
39    #[error("Invalid field format: {0}")]
40    FieldFormat(String),
41    ///Logical - when parsed data is logically inconsistent(for ex. wrong match status)
42    #[error("Logical inconsistency: {0}")]
43    Logical(String),
44}
45
46///Parses one line of football match data into a Match struct.
47///Each line should follow the format:
48///YYYY-MM-DD; HH:MM; HomeTeam - AwayTeam; ScoreOrMark; Stadium(optional); Status(optional)
49///It returns:
50///Ok(Match) if parsing succeeds  
51///Err(ParseError) if syntax, format, or logical errors occur
52pub fn parse_match_line(line: &str) -> Result<Match, ParseError> {
53    let pairs = MatchGrammar::parse(Rule::match_line, line)
54        .map_err(|_| ParseError::Syntax(format!("Invalid syntax in line: {}", line)))?;
55
56    let mut date = String::new();
57    let mut time = String::new();
58    let mut home_team = String::new();
59    let mut away_team = String::new();
60    let mut home_score: Option<u8> = None;
61    let mut away_score: Option<u8> = None;
62    let mut stadium: Option<String> = None;
63    let mut status: Option<String> = None;
64    let mut expected_status: Option<String> = None;
65
66    for pair in pairs.flatten() {
67        match pair.as_rule() {
68            Rule::date => {
69                let field = pair.as_str().trim();
70                NaiveDate::parse_from_str(field, "%Y-%m-%d")
71                    .map_err(|_| ParseError::FieldFormat(format!("Invalid date: {}", field)))?;
72                date = field.to_string();
73            }
74
75            Rule::time => {
76                let field = pair.as_str().trim();
77                time = field.to_string();
78            }
79
80            Rule::teams => {
81                let parts: Vec<&str> = pair.as_str().split('-').collect();
82                if parts.len() != 2 {
83                    return Err(ParseError::Syntax(format!(
84                        "Invalid teams format: {}",
85                        pair.as_str()
86                    )));
87                }
88                home_team = parts[0].trim().to_string();
89                away_team = parts[1].trim().to_string();
90            }
91
92            Rule::score_or_mark => {
93                let field = pair.as_str().trim().to_lowercase();
94
95                if let Some((home, away)) = field.split_once(':') {
96                    home_score = Some(home.trim().parse().map_err(|_| {
97                        ParseError::FieldFormat(format!("Invalid home score: {}", home.trim()))
98                    })?);
99
100                    away_score = Some(away.trim().parse().map_err(|_| {
101                        ParseError::FieldFormat(format!("Invalid away score: {}", away.trim()))
102                    })?);
103                    expected_status = Some("played".to_string());
104                } else if field == "-" {
105                    expected_status = Some("scheduled".to_string());
106                } else if field == "live" || field.ends_with('\'') {
107                    expected_status = Some("ongoing".to_string());
108                } else {
109                    return Err(ParseError::FieldFormat(format!(
110                        "Unknown score or mark: {}",
111                        field
112                    )));
113                }
114            }
115
116            Rule::stadium_name => {
117                let field = pair.as_str().trim().to_string();
118                let lower = field.to_lowercase();
119                let reserved_statuses = ["played", "scheduled", "ongoing", "live"];
120                if reserved_statuses.contains(&lower.as_str()) {
121                    status = Some(lower);
122                } else if !field.is_empty() {
123                    stadium = Some(field);
124                }
125            }
126
127            Rule::status => {
128                if status.is_none() {
129                    status = Some(pair.as_str().trim().to_lowercase());
130                }
131            }
132            _ => {}
133        }
134    }
135    if status.is_none() {
136        status = expected_status.clone();
137    }
138    validate_logical_mismatch(status.as_deref(), expected_status.as_deref())?;
139
140    Ok(Match {
141        date,
142        time,
143        home_team,
144        away_team,
145        home_score,
146        away_score,
147        stadium,
148        status,
149    })
150}
151///Validates whether the detected match status matches the expected one.
152fn validate_logical_mismatch(
153    status: Option<&str>,
154    expected_status: Option<&str>,
155) -> Result<(), ParseError> {
156    match (expected_status, status) {
157        (None, None) => Err(ParseError::Logical(
158            "Cannot determine match status".to_string(),
159        )),
160        (Some(expected), Some(actual)) if expected != actual => Err(ParseError::Logical(format!(
161            "Status mismatch: expected '{}', but found '{}'",
162            expected, actual
163        ))),
164        _ => Ok(()),
165    }
166}
